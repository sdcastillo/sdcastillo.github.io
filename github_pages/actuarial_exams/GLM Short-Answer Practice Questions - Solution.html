mae <- function(y, y_hat){
  mean(abs(y - y_hat))
}

fit_glm <- function(family_link = gaussian(link = "identity")){
  model = glm(formula = charges ~ age + gender + bmi + smoker + region + age_bucket + age_household_ratio + log_household_size + bmi*smoker, 
            family = family_link,
            data = train)
  pred = predict(model, test, type = "response")
  paste0(
    "family:", model$family$family, ", ",
    "link:", model$family$link, ", ",
    "test mae:", round(mae(pred, test$charges),0))
}
